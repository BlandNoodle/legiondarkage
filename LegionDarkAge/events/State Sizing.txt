country_event = {
	id = 700700
	title = "State Sizing"

	trigger = {
		tag = GHO
		NOT = { has_country_modifier = sizing_cooldown }
	}

	immediate = {
		add_country_modifier = {
			name = sizing_cooldown
			duration = 350
		}
	}

	option = { 
		any_country = { 
			any_owned = { 
				limit = {
					state_scope = { 
						total_pops = 10
						NOT = { total_pops = 50000 }
					}
				}
				add_core = TIN 
				remove_core = SMA 
				remove_core = MED 
				remove_core = LRG 
				remove_core = HUG 
			}
		}
		any_country = { 
			any_owned = { 
				limit = {
					state_scope = { 
						total_pops = 50000

				}
