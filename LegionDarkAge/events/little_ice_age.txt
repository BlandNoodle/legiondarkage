	picture = "port"

has_country_flag = pluage_jst
NOT = { has_country_flag = plague_measures}
NOT = { has_country_flag = had_pluage_jst }

	any_owned_province = {
			has_building = naval_base 
	}
	}
}

			has_building = naval_base 
		}

	add_country_modifier = { name = mitigated_plague duration = 360 }
	militancy = 3 
}

name = "This will all blow over"
	set_country_flag = plague_measures
	prestige = -15
consciousness = 3 
}

}
